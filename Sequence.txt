SET SERVEROUTPUT ON;

CREATE SEQUENCE customers_seq
	START WITH 1300
	INCREMENT BY   1
	NOCACHE
	NOCYCLE;	
SELECT customers_seq.NEXTVAL FROM DUAL;

CREATE SEQUENCE tables
	START WITH 0
	MINVALUE 0
	INCREMENT BY 2
	NOCACHE
	NOCYCLE;

INSERT INTO EMP VALUES (customers_seq.NEXTVAL,'cve',500);
INSERT INTO EMP VALUES (customers_seq.NEXTVAL,'rvb',459);
SELECT * FROM EMP;

DECLARE
	I NUMBER := 1;
BEGIN
	LOOP
		DBMS_OUTPUT.PUT_LINE(2||' * '||I||' = '||tables.NEXTVAL);
		I:=I+1;
		EXIT WHEN I=11;
	END LOOP;
END;
/

DROP SEQUENCE tables;
DROP SEQUENCE customers_seq;

COMMIT;
